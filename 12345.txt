# первое число от которого будем отнимать
число = 100
# здесь 5 чисел которые вводит пользователь
for i in range(5):
    #тут мы вводим числа которые будем отнимать (i+ 1)нужен лоя того чтобы отсчет шел не от 0 а от 1
    вычитание =  float(input(f"Введите число {i + 1}: "))
    # тут иы берем основное "число" и от него отнимаем "вычитание" 
    число -= вычитание
# Выводим получившийся остаток
print("Получившийся остаток:", число)

print("----------------------------")

# хранения суммый 
сумма = 0

# Запрос 10 чисел оь пользователя
for i in range(10):
    # Запрос числа с клавиатуры
    число = float(input(f"Введите число {i + 1}: "))
    
    # Добавляем число к сумме
    сумма += число

# Выводим сумму
print(f"Сумма введенных чисел: {сумма}")
